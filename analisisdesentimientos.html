"""
Crear una frase larga con palabras repetidas donde se expresen un sentimiento positivo

optimizando la frase y recorriendo la frase con un ciclo for. Con if contabilizar cu치ntas

# Listas de palabras positivas y negativas
pp = ["bueno", "bien", "feliz", "alegre", "contento", "maravilloso", "bonito"]
pn= ["feo", "triste", "solo", "aburrido", "malo", "horrible", "muerte"]

# Solicitar la frase
f = input("Ingrese una frase larga con sentimientos : ")

f = f.lower()
f=f.replace(",","").replace(":","").replace(".","").replace(".","")
palabras = f.split()

# Contadores y listas para mostrar palabras encontradas
positivas = 0
negativas = 0
palabras_positivas = []
palabras_negativas = []

for palabra in palabras:
    if palabra in pp:
        positivas += 1
        palabras_positivas.append(palabra)
    elif palabra in pn:
        negativas += 1
        palabras_negativas.append(palabra)

print("An치lisis de sentimientos:")
print("Palabras positivas encontradas:", positivas,)
print(f"Palabras negativas encontradas:", negativas,)

if positivas > negativas:
    print("Es un Texto Positivo")
elif negativas > positivas:
    print("Es un Texto Negativo")
